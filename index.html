function dibujarTablaYCalcularGlobal() { const tBody = document.getElementById('cuerpoTablaMeses'); tBody.innerHTML = ''; let saldoGlobalMins = 0; // Cambiamos la lista fija por un bucle que recorre desde Enero (0) hasta el mes que tienes activo en el historial for (let mIdx = 0; mIdx <= historialMesActivo; mIdx++) { const año = historialAñoActivo; const { totalMinutosTeoricos, diasMapeados } = obtenerDiasLaborablesYTeoricosMes(año, mIdx); let minutosFichadosMes = 0; let tieneRegistros = false; // Bandera para no mostrar meses vacíos del pasado si no quieres for (const fechaStr in diasMapeados) { const reg = baseDatos[fechaStr]; if (reg) { tieneRegistros = true; if (reg.incidencia) { minutosFichadosMes += diasMapeados[fechaStr]; } else { minutosFichadosMes += calcularMinutosDeUnDia(fechaStr); } } } // Si el mes tiene teóricos (es laborable) o tiene registros guardados, lo sumamos al saldo global let saldoMesMins = minutosFichadosMes - totalMinutosTeoricos; // Solo acumulamos en la bolsa global si el mes ya ha concluido o si tiene actividad saldoGlobalMins += saldoMesMins; const tr = document.createElement('tr'); const colorSaldo = saldoMesMins >= 0 ? 'var(--success-color)' : 'var(--danger-color)'; tr.innerHTML = ` ${mesesNombres[mIdx]} ${año} ${tabularMinsStr(minutosFichadosMes)} ${tabularMinsStr(totalMinutosTeoricos)} ${tabularMinsStr(saldoMesMins)} `; tBody.appendChild(tr); } const globalDiv = document.getElementById('saldoGlobalTotal'); globalDiv.innerText = tabularMinsStr(saldoGlobalMins); globalDiv.style.color = saldoGlobalMins >= 0 ? 'var(--success-color)' : 'var(--danger-color)'; }
