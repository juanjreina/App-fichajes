function dibujarTablaYCalcularGlobal() { const cuerpo = document.getElementById('cuerpoTablaMeses'); cuerpo.innerHTML = ''; // BLINDADO: Estos meses quedan cerrados con los saldos oficiales que coinciden con Atalaya let mesesEstaticos = [ { id: 0, nombre: 'Enero', fichado: 75*60 + 57, teorico: 95*60 + 12, saldo: (75*60 + 57) - (95*60 + 12) }, { id: 1, nombre: 'Febrero', fichado: 117*60 + 5, teorico: 112*60 + 0, saldo: (117*60 + 5) - (112*60 + 0) }, { id: 2, nombre: 'Marzo', fichado: 96*60 + 17, teorico: 106*60 + 24, saldo: (96*60 + 17) - (106*60 + 24) }, { id: 3, nombre: 'Abril', fichado: 79*60 + 2, teorico: 95*60 + 12, saldo: (79*60 + 2) - (95*60 + 12) }, { id: 4, nombre: 'Mayo', fichado: 99*60 + 8, teorico: 100*60 + 48, saldo: (99*60 + 8) - (100*60 + 48) } ]; let acumuladoSaldosPasados = 0; mesesEstaticos.forEach(m => { acumuladoSaldosPasados += m.saldo; inyectarFilaTablaResumen(m.nombre, m.fichado, m.teorico, m.saldo); }); // DINÁMICO: Cómputo de Junio en adelante automatizado con las reglas de Atalaya let minsTeoricosJunio = 0; let minsFichadosJunio = 0; const fechaHoy = new Date(); // Si estamos en Junio de 2026, cuenta hasta el día de hoy; si no, cuenta el mes entero (30 días) const diaHoyJunio = (fechaHoy.getFullYear() === 2026 && fechaHoy.getMonth() === 5) ? fechaHoy.getDate() : 30; for (let d = 1; d <= 30; d++) { let fechaKey = `2026-06-${String(d).padStart(2, '0')}`; let reg = baseDatos[fechaKey]; let dObj = new Date(2026, 5, d); let diaSemana = dObj.getDay(); let esFinde = (diaSemana === 0 || diaSemana === 6); // 1. Teórico de Lunes a Viernes: Suma siempre para los días pasados o el actual if (!esFinde && d <= diaHoyJunio) { minsTeoricosJunio += obtenerMinutosTeoricosDia(fechaKey); } // 2. Fichado Real: Suma lo que se haya trabajado ese día minsFichadosJunio += calcularMinutosDeUnDia(fechaKey); // 3. Compensación de Festivos (Día Especial) de Lunes a Viernes if (!esFinde && reg && reg.incidencia === true && d <= diaHoyJunio) { minsFichadosJunio += obtenerMinutosTeoricosDia(fechaKey); } } let saldoJunio = minsFichadosJunio - minsTeoricosJunio; inyectarFilaTablaResumen('Junio', minsFichadosJunio, minsTeoricosJunio, saldoJunio); // Bolsa Total = Histórico bloqueado + Junio automático limpio let saldoGlobalMinutos = acumuladoSaldosPasados + saldoJunio; const panelGlobal = document.getElementById('saldoGlobalTotal'); if (saldoGlobalMinutos > 0) { panelGlobal.style.color = 'var(--success-color)'; } else if (saldoGlobalMinutos < 0) { panelGlobal.style.color = 'var(--danger-color)'; } else { panelGlobal.style.color = 'var(--text-color)'; } panelGlobal.innerText = tabularMinsStr(saldoGlobalMinutos); }
